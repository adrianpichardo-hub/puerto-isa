// --- LÓGICA MEJORADA PARA ADRIAN (ISA) --- const CONFIG = { bargePrefixes: ['HB', 'LHG', 'ORE', 'CIE', 'MAG', 'TBN', 'SCF', 'NAV', 'BCC'], defaultObservation: 'DESCARGA A PILA', agencyEvents: [ { id: 'h_arr', label: (rem, lug) => `${rem} + 12 barcazas amarra en ${lug}.` }, { id: 'h_ali', label: () => 'Carta de alistamiento extendida.' }, { id: 'h_aut', label: () => 'Autoridades y agencia a bordo.' }, { id: 'h_pla', label: () => 'Libre plática otorgada.' } ] }; const bargeRegex = new RegExp(`(${CONFIG.bargePrefixes.join('|')})\\s*(\\d+)`, 'i'); const timeRegex = /\b(\d{1,2}:\d{2})\b/g; function procesarLog() { const logText = document.getElementById('logInput').value.trim(); const fecha = document.getElementById('v_fec').value.trim(); const rem = document.getElementById('v_rem').value.trim() || 'REMOLCADOR'; const lugar = document.getElementById('v_lug').value.trim() || 'LUGAR'; const tbody = document.getElementById('bodySOF'); tbody.innerHTML = ''; const rows = []; // 1. Eventos fijos de agencia CONFIG.agencyEvents.forEach(event => { const hora = document.getElementById(event.id)?.value?.trim(); if (hora) { const obs = typeof event.label === 'function' ? event.label(rem, lugar) : event.label; rows.push({ fecha, barcaza: '---', desde: hora, hasta: '', obs, css: '' }); } }); // 2. Parseo del log línea por línea let barcazaActual = '---'; const lineas = logText.split(/\r?\n/); lineas.forEach(linea => { linea = linea.trim(); if (linea.length < 4) return; // Detectar cambio de barcaza const bargeMatch = linea.match(bargeRegex); if (bargeMatch) barcazaActual = bargeMatch[0].toUpperCase(); // Extraer horarios const tiempos = [...linea.matchAll(timeRegex)].map(m => m[1]); if (!tiempos.length) return; // LIMPIEZA DE OBSERVACIÓN - LA CLAVE let obsOriginal = linea.replace(timeRegex, '').replace(bargeRegex, '').trim().toUpperCase(); let obsFinal = obsOriginal; // Diccionario de eventos ISA if (obsOriginal.includes("DRAFT FINAL") || obsOriginal.includes("FINALIZA")) { obsFinal = `FINALIZA DESCARGA / DRAFT FINAL - BCZA ${barcazaActual}`; rows.push({ fecha, barcaza: barcazaActual, desde: tiempos[0], hasta: '', obs: obsFinal, css: 'row-fix' }); return; // Evita duplicar la línea } if (obsOriginal.includes("INICIA") || obsOriginal.includes("COMIENZA")) { obsFinal = `INICIA DESCARGA DE BARCAZA ${barcazaActual}`; } else if (obsOriginal.includes("ABRE") || obsOriginal.includes("OPENING")) { obsFinal = `APERTURA DE BODEGAS / PREPARATIVOS`; } else if (obsOriginal.length < 3) { obsFinal = CONFIG.defaultObservation; } rows.push({ fecha, barcaza: barcazaActual, desde: tiempos[0], hasta: tiempos[1] || '', obs: obsFinal, css: '' }); }); // Ordenar y Renderizar rows.sort((a, b) => a.desde.localeCompare(b.desde)); rows.forEach(r => { const tr = document.createElement('tr'); if (r.css) tr.className = r.css; tr.innerHTML = `${r.fecha}${r.barcaza}${r.desde}${r.hasta}${r.obs}`; tbody.appendChild(tr); }); document.getElementById('btnXLS').style.display = rows.length > 0 ? 'inline-block' : 'none'; }
